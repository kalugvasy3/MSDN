It is SQLite Editor. (For reducing size of a file, was deleted all SQLite package from the project. VS will have to restore all package from NuGet automatically.)

You just need to use mouse … everything is doing automatically …  but you can do something and manually…

One cool thing - point the mouse to the Microsoft Flag and using mouse wheel change scale.

“Left mouse click change style of window – Tool, None, Windows Application.

“Right mouse click “– restore to original size and style.

It works perfectly if you Run this application on higher resolution display

(VS 2015 should not RUN like administrator – drop file – logic will not be working)

What do you learn from this project?

  1. How to SQLite.
  2. How to UserControl – event, generate, subscribe, how to return data to parent window …
  3. How to scale current window application only…
  4. How to dynamically build SQL code…
  5. How to convert XML, CSV to DataSet.
  6. How to Dynamically add new TextBox, Tab, DataGrid, etc …  
  7. SQLite class which you can reuse in different type of application - it PASS VERACODE test …

 

You can – Open existing database.

You can create new database (Just insert new name in connection string and [Connect])

If database is NEW – [Connection Button will be pink, see below screenshot]

 

You can import different database (in XML format) to SQLite databases, just use [Import] button and drop XML file in [DROP HERE] area or just paste some text to textbox.

 

It is new database … File will be created in current folder …

 

 

If databases exist – it will be green … Project include test databases – Northwind.sl3

File extension for SQLite lite is not important … 

 

 

How to IMPORT data and create new table dynamically …

 

 

Select [Import] and drop file (example show for csv file). Table name will be same like file name.

Select [TRANSFER DATA]

 

 

Now you will be able to create SQL code for TABLE [CRETE TABLE]. This create in new TAB “Create …” just verify if needed and [Run].  Code created base on csv data. 

 

Next step [CREATE INSERT] and [RUN].

If you skip [CREATE INSERT] step, data will be appnded to table....

 

Reconnect to database [Connect] and verify.  Go to [Schema] or [Select]

        private void btnStyle_MouseWheel(object sender, MouseWheelEventArgs e)
        {
            if (ucScale.ScaleX == 1.0)
            {
                origH = userMainControl.Height;
                origW = userMainControl.Width;
            }

            userMainControl.Width = origW * ucScale.ScaleX;
            userMainControl.Height = origH * ucScale.ScaleY;

            if (e.Delta > 0)
            {

                ucScale.ScaleX = ucScale.ScaleX + 0.1;
                ucScale.ScaleY = ucScale.ScaleY + 0.1;

                if (ucScale.ScaleX >= 1.7)
                {
                    ucScale.ScaleX = 1.7;
                    ucScale.ScaleY = 1.7;
                }
            };

            if (e.Delta < 0)
            {
                ucScale.ScaleX = ucScale.ScaleX - 0.1;
                ucScale.ScaleY = ucScale.ScaleY - 0.1;

                if (ucScale.ScaleX <= 0.7)
                {
                    ucScale.ScaleX = 0.7;
                    ucScale.ScaleY = 0.7;
                }
            };
        }

        private void listOfTables_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
            TabItem ti = (TabItem)tabPlaceHolder.SelectedItem;

            // Insert Table Name ...
            ListBox lb = (ListBox)sender;
            DataRowView drv = (DataRowView)lb.SelectedValue;
            if (drv == null) return;
            // Put Text to current TEXT BOX.  INSERT/UPDATE
            string strTableName = drv["tbl_name"].ToString().Trim();

            //strTextOfCommand  common value...
            string strRun = strTextOfCommand.Replace("TABLE_NAME", strTableName);

            if (strTextOfCommand.IndexOf("ALTER") >= 0)
            {
                strRun = helper.BringAlterSql(txtConnectionString.Text, strTableName);
            }
            if (strTextOfCommand.IndexOf("SELECT") >= 0)
            {
                strRun = helper.BringSelectSql(txtConnectionString.Text, strTableName);
            }

            if (strTextOfCommand.IndexOf("INSERT") >= 0)
            {
                strRun = helper.BringInsertSql(txtConnectionString.Text, strTableName);
            }

            if (strTextOfCommand.IndexOf("UPDATE") >= 0)
            {
                strRun = helper.BringUpdateSql(txtConnectionString.Text, strTableName);
            }

            if (strTextOfCommand.IndexOf("DELETE") >= 0)
            {
                strRun = helper.BringDeleteSql(txtConnectionString.Text, strTableName);
            }

            ((TextBox)ti.Content).Text = strRun;

            listOfTables.SelectedIndex = -1;
            listOfTables.Visibility = Visibility.Collapsed;
        }

        void newImport_readySQLDataInsert(object sender)

            p.Topmost = true;

            TextBox tb = (TextBox)ti.Content;

            p.Topmost = true;  
        }

        private DependencyObject GetTopLevelControl(DependencyObject control)
        {
            DependencyObject tmp = control;
            DependencyObject parent = null;
            while ((tmp = VisualTreeHelper.GetParent(tmp)) != null)
            {
                parent = tmp;
            }
            return parent;
        }

        private void listOfTables_SelectionChanged(object sender, SelectionChangedEventArgs e) 
        { 
            TabItem ti = (TabItem)tabPlaceHolder.SelectedItem; 
 
            // Insert Table Name ... 
            ListBox lb = (ListBox)sender; 
            DataRowView drv = (DataRowView)lb.SelectedValue; 
            if (drv == null) return; 
            // Put Text to current TEXT BOX.  INSERT/UPDATE 
            string strTableName = drv["tbl_name"].ToString().Trim(); 
 
            //strTextOfCommand  common value... 
            string strRun = strTextOfCommand.Replace("TABLE_NAME", strTableName); 
 
            if (strTextOfCommand.IndexOf("ALTER") >= 0) 
            { 
                strRun = helper.BringAlterSql(txtConnectionString.Text, strTableName); 
            } 
            if (strTextOfCommand.IndexOf("SELECT") >= 0) 
            { 
                strRun = helper.BringSelectSql(txtConnectionString.Text, strTableName); 
            } 
 
            if (strTextOfCommand.IndexOf("INSERT") >= 0) 
            { 
                strRun = helper.BringInsertSql(txtConnectionString.Text, strTableName); 
            } 
 
            if (strTextOfCommand.IndexOf("UPDATE") >= 0) 
            { 
                strRun = helper.BringUpdateSql(txtConnectionString.Text, strTableName); 
            } 
 
            if (strTextOfCommand.IndexOf("DELETE") >= 0) 
            { 
                strRun = helper.BringDeleteSql(txtConnectionString.Text, strTableName); 
            } 
 
            ((TextBox)ti.Content).Text = strRun; 
 
            listOfTables.SelectedIndex = -1; 
            listOfTables.Visibility = Visibility.Collapsed; 
        }

        private DependencyObject GetTopLevelControl(DependencyObject control) 
        { 
            DependencyObject tmp = control; 
            DependencyObject parent = null; 
            while ((tmp = VisualTreeHelper.GetParent(tmp)) != null) 
            { 
                parent = tmp; 
            } 
            return parent; 
        }
